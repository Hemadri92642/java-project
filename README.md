# Sample java-project 


# Clone this repository
$ git clone https://github.com/Hemadri92642/java-project.git

# Go into the repository
$ cd java-project

$ javac MyJavaFile.java

$ jar cmvf MANIFEST.MF myJarFile.jar MyJavaFile.class

# Test jar file
$ java -jar myJarFile.jar
